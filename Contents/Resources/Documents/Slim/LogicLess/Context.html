def lambda(name)
  scope.lambda(name) do |*dict|
    if dict.empty?
      yield
    else
      new_scope do
        dict.inject('') do |result, d|
          scope.dict = d
          result << yield
        end
      end
    end
  end
end

def section(name)
  if dict = scope[name]
    if !dict.respond_to?(:has_key?) && dict.respond_to?(:each)
      new_scope do
        dict.each do |d|
          scope.dict = d
          yield
        end
      end
    else
      new_scope(dict) { yield }
    end
  end
end
