def initialize(opts = {})
  super
  case options[:tr_mode]
  when :static
    @translator = StaticTranslator.new(:tr_fn => options[:tr_fn])
  when :dynamic
    @translator = DynamicTranslator.new(:tr_fn => options[:tr_fn])
  else
    raise ArgumentError, "Invalid translator mode #{options[:tr_mode].inspect}"
  end
end

  text
end
